const calculateTrend = (current, previous) => {
  const cur = toNumber(current);
  const prev = toNumber(previous);
  if (prev === 0 && cur === 0) return 0;
  if (prev === 0) return cur > 0 ? 100 : -100;
  return Number((((cur - prev) / Math.abs(prev)) * 100).toFixed(1));
};
